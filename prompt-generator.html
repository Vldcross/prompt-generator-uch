  <h2>Генератор промптов для MidJourney</h2>

  <form id="prompt-form">

    <label for="conditions">Задаем условия:</label><br>

    <input type="text" id="conditions" name="conditions" placeholder="Например: Студийное фото, парное фото преподавателей информатики из России"><br><br>


    <label for="emotions">Перечисляем эмоции:</label><br>

    <input type="text" id="emotions" name="emotions" placeholder="Например: естественные эмоции, улыбаются"><br><br>


    <label for="characters">Задаем параметры персонажей на фото:</label><br>

    <textarea id="characters" name="characters" rows="4" placeholder="Например: женщина стоит перед мужчиной, руки скрещены, мужчина 45 лет с легкой щетиной..."></textarea><br><br>


    <label for="rendering">Задаем параметры рендеринга:</label><br>

    <textarea id="rendering" name="rendering" rows="3" placeholder="Например: реалистичное фото, текстура кожи выражена..."></textarea><br><br>


    <label for="year">Обозначаем год:</label><br>

    <input type="text" id="year" name="year" placeholder="Например: 2024"><br><br>


    <label for="background">Задаем параметры фона:</label><br>

    <input type="text" id="background" name="background" placeholder="Например: фон фиолетовое фотополотно хромакей"><br><br>


    <button type="button" id="generate-button">Сгенерировать промпт</button>

  </form>


  <h3>Результат:</h3>

  <p id="result"></p>


  <script>

    document.getElementById('generate-button').addEventListener('click', function () {

      const conditions = document.getElementById('conditions').value || "Без условий";

      const emotions = document.getElementById('emotions').value || "Без эмоций";

      const characters = document.getElementById('characters').value || "Без описания персонажей";

      const rendering = document.getElementById('rendering').value || "Без рендеринга";

      const year = document.getElementById('year').value || "Без указания года";

      const background = document.getElementById('background').value || "Без фона";


      const prompt = `${conditions}, ${emotions}, ${characters}, ${rendering}, ${year} год, ${background}`;

      document.getElementById('result').textContent = prompt;

    });

  </script>
